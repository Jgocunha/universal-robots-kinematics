52 [[nodiscard]] pose divideByConst(float constant) const
53 {
54 return {m_pos[0] / constant, m_pos[1] / constant, m_pos[2] / constant,
55 m_eulerAngles[0] / constant, m_eulerAngles[1] / constant, m_eulerAngles[2] / constant};

58 [[nodiscard]] pose subtract(const pose& other) const
59 {
60 return {m_pos[0] - other.m_pos[0],
61 m_pos[1] - other.m_pos[1],
62 m_pos[2] - other.m_pos[2],
63 m_eulerAngles[0] - other.m_eulerAngles[0],
64 m_eulerAngles[1] - other.m_eulerAngles[1],
65 m_eulerAngles[2] - other.m_eulerAngles[2]};
66 }
